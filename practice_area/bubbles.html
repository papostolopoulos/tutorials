Classic bubble

        // Typical bubble
          let innerBubble = window['innerBubble'];
            innerBubble.addEventListener('click', ()=>{
              console.log("innerBubble clicked");
              innerBubble.style.backgroundColor = "lime";
            });

          let middleBubble = window['middleBubble'];
            middleBubble.addEventListener('click', ()=>{
              console.log("middleBubble clicked");
              middleBubble.style.backgroundColor = "pink";
            });

          let outerBubble = window['outerBubble'];
            outerBubble.addEventListener('click', ()=>{
              console.log("outerBubble clicked");
              outerBubble.style.backgroundColor = "gray";
            });

Reverse bubble (Timeout set so the effect is visiually better to understand)

        // reversed bubble
          let innerBubbleReverse = window['innerBubbleReverse'];
            innerBubbleReverse.addEventListener('click', ()=>{
              setTimeout(()=>{
                console.log("innerBubbleReverse clicked");
                innerBubbleReverse.style.backgroundColor = "lime";
              }, 1000);
            }, true);

          let middleBubbleReverse = window['middleBubbleReverse'];
            middleBubbleReverse.addEventListener('click', function(){
              setTimeout(()=>{
                console.log("middleBubbleReverse clicked");
                middleBubbleReverse.style.backgroundColor = "pink";
              }, 500);
            }, true);

          let outerBubbleReverse = window['outerBubbleReverse'];
            outerBubbleReverse.addEventListener('click', function(){
              console.log("outerBubbleReverse clicked");
              outerBubbleReverse.style.backgroundColor = "gray";
            }, true);

Stop Propagation

        // Stop Propagation
        let innerStopPropagation = window["innerStopPropagation"];
        innerStopPropagation.addEventListener("click", (event)=>{
          event.stopPropagation();
          console.log("innerStopPropagation clicked");
          innerStopPropagation.style.background = "lime";
        });

        let middleStopPropagation = window["middleStopPropagation"];
        middleStopPropagation.addEventListener("click", (event)=>{
          event.stopPropagation();
          console.log("middleStopPropagation clicked");
          middleStopPropagation.style.background = "pink";
        });

        let outerStopPropagation = window["outerStopPropagation"];
        outerStopPropagation.addEventListener("click", (event)=>{
          event.stopPropagation();
          console.log("outerStopPropagation clicked");
          outerStopPropagation.style.background = "gray";
        });
